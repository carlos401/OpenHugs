CLASE Algoritmos {
	FUNCION eratostenes (NUMERO n) RETORNA Vector:LOGICO {
		Vector:LOGICO respuesta
		NUMERO i := 0
		MIENTRAS (i <= n){
			respuesta.agregar(i,VERDADERO)
			i++
		}
		NUMERO limite := raiz(n) ;para control del algoritmo
		i := 2 ;reutilización de la variable i declarada anteriormente
		MIENTRAS (i <= limite)
			SI (respuesta.obtener(i)) {
				NUMERO j := i
				MIENTRAS (j <= n/i) {
					respuesta.agregar(i*j, FALSO)
					j++
				}
			}
			i++
		RETORNAR respuesta
	};fin de la función
} ;fin de la clase
