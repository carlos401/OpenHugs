CLASE Algoritmos {
	FUNCION triangulo (PALABRA letra , NUMERO niveles){
		NUMERO i := 1
		MIENTRAS ( i ) {
			NUMERO e := 1
			MIENTRAS (e <= niveles - i) {
				imprimir(" ")
				e++
			}
			NUMERO j := 0
			CICLO MIENTRAS (j < i*2-1) {
				imprimir(letra)
				j++
			}
			i++
		} ;fin del primer ciclo	
	} ;fin de la función
} ;fin de la clase
